# 피타고라스 정리 공식
    a^2+b^2=c^2

# 높이 및 너비가 값이 아닌 비율로 제시되었을 경우, 비율을 구하기 위한 x 계산 공식

    d,h,w=map(int,input().split()) // d:대각선, h:높이, w:너비

    // d**2 = (h*x)**2 + (w*x)**2의 식을 정리하면
    x = (d**2 / (h**2 + w**2)) **0.5

    // 출력은 높이 h*x와 너비 w*x를 정수로!
    print(int(h*x), end=" ") //end=" " : 공백 한칸으로 구분
    print(int(w*x))
